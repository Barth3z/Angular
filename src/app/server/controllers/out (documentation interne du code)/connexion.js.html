const UserModel = require('../models/user');
const jwt = require('jsonwebtoken');



const loginCtrl = {};

/**
 * @constructor
 * @param {*} req - it send a request.
 * @param {*} res - it try to seek a response from the request.
 */

loginCtrl.logIn = async (req, res) => {
  const {mail, mdp} = req.body

  try {
    console.log("voici : " + mail + mdp);
    const user = await UserModel.login(mail, mdp); //Voir dans la BD si cet utilisateur existe
    res.status(201).json(
      {
        //status: 'trouvé'
        user: user._id
      }
    )
  }
  catch(err) {
    res.status(240).json({status: "Mot de passe ou mail incorrect"});
  }
};

module.exports = loginCtrl;
