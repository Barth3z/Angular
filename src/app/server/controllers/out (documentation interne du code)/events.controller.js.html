const EventsModel = require('../models/events');
const mongoose = require('mongoose');

const eventsCtrl = {};


/**
 * Represents the creation of an event in the calendar of the doctor.
 * @constructor   
 * @param {*} req - it send a request.
 * @param {*} res - it try to seek a response from the request.
 * @param {const} unEvent - it create an object that will be save in the database if it's correct.
 */
eventsCtrl.createEvent = async (req, res) => {
  const unEvent = new EventsModel(req.body);
    try {
      await unEvent.save();
      res.status(201).json({
        status: 'Rendez-vous enregistré'
      });
    }
    catch(err) {
      res.status(204).json({
        status: 'Evènement existant'
      });
    }
}

/**
 * Represents a print of the collection from the database.
 * @constructor 
 * @param {*} req - it send a request.
 * @param {*} res - it try to seek a response from the request.
 * @param {const} events - it print the collection from the database but without the _id and the __v of the user.  
 */
eventsCtrl.getEvents = async(req, res) => {
  const events = await EventsModel.find().select('-__v').select('-_id');
  res.status(201).json(events);
}

/**
 * Represents a deletion of an event in the calendar.

 * @param {*} res - it try to seek a response from the request.
 * @param {const} evenement - It help to find a specific  start of an event to delete it.
 * @param {const} unEvent - It try to find a specific id that has been wrote by the user that belongs to the database, in order to remove the event.
 */
eventsCtrl.deleteEvent = async(req, res) => {
  const evenement = await EventsModel.findOne({start: req.params.title});
  console.log(evenement);
  const unEvent = await EventsModel.findByIdAndRemove(evenement);
  console.log(unEvent);
  res.json({
    status: 'Supprimé'
  });
}

module.exports = eventsCtrl;
