/**
 * Represent a new meeting.
 * @constructor
 * @param {*} req - it send a request.
 * @param {*} res - it try to seek a response from the request.
 * @var {const} rdv - Creating an object Rdv.
 */
rdvCtrl.postNewRdv = async(req, res) => {
  const rdv = new Rdv(req.body);
  try {
    await rdv.save();
    res.status(201).json({
      status: 'ok'
    });
  }
  catch(err) {
    res.status(204).json({
      err,
      status: 'déjà pris !'
    })
  }
};
